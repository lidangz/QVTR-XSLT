CREATE SCHEMA Library；
CREATE TABLE Library.Copy (
Copy_tid integer not null ,
barcode varchar(60),
available tinyint,
Copy_borrowedBy_Loan_tid integer,
Copy_copyOf_Publication_tid integer,
Copy_heldBy_Reservation_tid integer,
CONSTRAINT Copy_pk PRIMARY KEY ( Copy_tid ))；
CREATE TABLE Library.Reservation (
Reservation_tid integer not null ,
Reservation_hold_Copy_tid integer,
Reservation_reserve_Publication_tid integer,
Reservation_makedBy_Reader_tid integer,
CONSTRAINT Reservation_pk PRIMARY KEY ( Reservation_tid ))；
CREATE TABLE Library.Reader (
Reader_tid integer not null ,
eligible tinyint,
copyNum integer,
name varchar(60),
addr_city varchar(60),
addr_eMail varchar(60),
addr_phone varchar(60),
CONSTRAINT Reader_pk PRIMARY KEY ( Reader_tid ))；
CREATE TABLE Library.Loan (
Loan_tid integer not null ,
dueDate varchar(60),
Loan_borrow_Copy_tid integer,
Loan_takenBy_Reader_tid integer,
CONSTRAINT Loan_pk PRIMARY KEY ( Loan_tid ))；
CREATE TABLE Library.Publication (
Publication_tid integer not null ,
name varchar(60),
info varchar(60),
CONSTRAINT Publication_pk PRIMARY KEY ( Publication_tid ))；
ALTER TABLE Library.Copy ADD CONSTRAINT Copy_borrowedBy_Loan FOREIGN KEY ( Copy_borrowedBy_Loan_tid) REFERENCES Library.Loan；
ALTER TABLE Library.Copy ADD CONSTRAINT Copy_copyOf_Publication FOREIGN KEY ( Copy_copyOf_Publication_tid) REFERENCES Library.Publication；
ALTER TABLE Library.Copy ADD CONSTRAINT Copy_heldBy_Reservation FOREIGN KEY ( Copy_heldBy_Reservation_tid) REFERENCES Library.Reservation；
ALTER TABLE Library.Reservation ADD CONSTRAINT Reservation_hold_Copy FOREIGN KEY ( Reservation_hold_Copy_tid) REFERENCES Library.Copy；
ALTER TABLE Library.Reservation ADD CONSTRAINT Reservation_reserve_Publication FOREIGN KEY ( Reservation_reserve_Publication_tid) REFERENCES Library.Publication；
ALTER TABLE Library.Reservation ADD CONSTRAINT Reservation_makedBy_Reader FOREIGN KEY ( Reservation_makedBy_Reader_tid) REFERENCES Library.Reader；
ALTER TABLE Library.Loan ADD CONSTRAINT Loan_borrow_Copy FOREIGN KEY ( Loan_borrow_Copy_tid) REFERENCES Library.Copy；
ALTER TABLE Library.Loan ADD CONSTRAINT Loan_takenBy_Reader FOREIGN KEY ( Loan_takenBy_Reader_tid) REFERENCES Library.Reader；
